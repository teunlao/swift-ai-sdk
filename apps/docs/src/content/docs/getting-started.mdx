# Getting Started

Install (SwiftPM)

```bash
git clone https://github.com/teunlao/swift-ai-sdk
cd swift-ai-sdk
swift build && swift test
```

Add to Package.swift

```swift
// Package.swift (example)
dependencies: [
  .package(url: "https://github.com/teunlao/swift-ai-sdk", branch: "main")
],
targets: [
  .target(
    name: "YourApp",
    dependencies: [
      .product(name: "SwiftAISDK", package: "swift-ai-sdk"),
      .product(name: "AISDKProvider", package: "swift-ai-sdk"),
      .product(name: "AISDKProviderUtils", package: "swift-ai-sdk")
    ]
  )
]
```

Basic example (generateText)

```swift
import SwiftAISDK
import AISDKProvider

@main
struct Demo {
  static func main() async throws {
    let model = OpenAICompatible.TextModel(
      id: "gpt-4o-mini",
      baseURL: URL(string: "https://api.openai.com/v1")!,
      apiKey: ProcessInfo.processInfo.environment["OPENAI_API_KEY"] ?? ""
    )

    let result = try await generateText(
      model: model,
      input: "Write a short poem about the wind"
    )
    print(result.text)
  }
}
```

Streaming output (streamText)

```swift
let stream = try await streamText(
  model: model,
  input: "List 3 README ideas"
)
for try await event in stream {
  if case let .textDelta(delta) = event { print(delta, terminator: "") }
}
```

Structured data (generateObject)

```swift
import AISDKProviderUtils

struct Repo: Codable, Sendable { let name: String; let stars: Int }
let schema: Schema<Repo> = .codable(Repo.self)

let obj = try await generateObject(model: model, input: "Repo name and stars", schema: schema)
print(obj)
```
